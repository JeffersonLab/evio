80  FileCloserV4(int poolSize) : size(poolSize) {
81  // Create a work object to keep the io_context busy
82  boost::asio::io_context::work work(ioContext);
83 
84  // Add multiple threads to pool
85  for (int i = 0; i < poolSize; ++i) {
86  threadPool.create_thread([&]() {
87  ioContext.run();
88  });
89  }
90  }
91 
92  void close() {
93  threadPool.join_all();
94  }

96  void closeAsyncFile(std::shared_ptr<std::fstream> afc,
97  std::shared_ptr<std::future<void>> future) {
98  ioContext.post([=]() {
99  try {
100  // There may be a simultaneous write in progress,
101  // wait for it to finish.
102  if (future != nullptr && future->valid()) {
103  // Wait for last write to end before we continue
104  future->get();
105  }
106  }
107  catch (std::exception & e) {}
108 
109  afc->close();
110  });
111  }
