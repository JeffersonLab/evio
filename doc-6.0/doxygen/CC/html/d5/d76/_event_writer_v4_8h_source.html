104  void closeAsyncFile(std::shared_ptr<std::fstream> & afc,
105  std::shared_ptr<std::future<void>> future) {
106  ioContext.post([=]() {
107  try {
108  // There may be a simultaneous write in progress,
109  // wait for it to finish.
110  if (future != nullptr && future->valid()) {
111  // Wait for last write to end before we continue
112  future->get();
113  }
114  }
115  catch (std::exception & e) {}
116 
117  afc->close();
118  });
119  }
