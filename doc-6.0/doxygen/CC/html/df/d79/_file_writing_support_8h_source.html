185  void writeTrailerToFile() {
186 
187  // Keep track of where we are right now which is just before trailer
188  uint64_t trailerPosition = bytesWrittenToFile;
189 
190  // If we're NOT adding a record index, just write trailer
191  if (!writeIndx) {
192  try {
193  // hdrBuffer is only used in this method
194  hdrBuffer.position(0).limit(RecordHeader::HEADER_SIZE_BYTES);
195  RecordHeader::writeTrailer(hdrBuffer, 0, recordNum, nullptr);
196  }
197  catch (EvioException &e) {/* never happen */}
198 
199  afChannel->write(reinterpret_cast<char *>(hdrArray),
200  RecordHeader::HEADER_SIZE_BYTES);
201  if (afChannel->fail()) {
202  throw EvioException("error writing to file");
203  }
204  }
205  else {
206  // Write trailer with index
207 
208  // How many bytes are we writing here?
209  uint32_t bytesToWrite = RecordHeader::HEADER_SIZE_BYTES + 4 * recLengths->size();
210 
211  // Make sure our array can hold everything
212  if (hdrBufferBytes < bytesToWrite) {
213  hdrBuffer = ByteBuffer(bytesToWrite);
214  hdrArray = hdrBuffer.array();
215  }
216  hdrBuffer.limit(bytesToWrite).position(0);
217 
218  // Place data into hdrBuffer - both header and index
219  try {
220  RecordHeader::writeTrailer(hdrBuffer, (size_t) 0, recordNum, recLengths);
221  }
222  catch (EvioException &e) {/* never happen */}
223  afChannel->write(reinterpret_cast<char *>(hdrArray), bytesToWrite);
224  if (afChannel->fail()) {
225  throw EvioException("error writing to file");
226  }
227  }
228 
229  // Update file header's trailer position word
230  if (!byteOrder.isLocalEndian()) {
231  trailerPosition = SWAP_64(trailerPosition);
232  }
233 
234  afChannel->seekg(FileHeader::TRAILER_POSITION_OFFSET);
235  afChannel->write(reinterpret_cast<char *>(&trailerPosition), sizeof(trailerPosition));
236  if (afChannel->fail()) {
237  throw EvioException("error writing to file");
238  }
239 
240  // Update file header's bit-info word
241  if (writeIndx) {
242  uint32_t bitInfo = fHeader.setBitInfo(fHeader.hasFirstEvent(),
243  fHeader.hasDictionary(),
244  true);
245  if (!byteOrder.isLocalEndian()) {
246  bitInfo = SWAP_32(bitInfo);
247  }
248  afChannel->seekg(FileHeader::BIT_INFO_OFFSET);
249  afChannel->write(reinterpret_cast<char *>(&bitInfo), sizeof(bitInfo));
250  if (afChannel->fail()) {
251  throw EvioException("error writing to file");
252  }
253  }
254 
255  // Update file header's record count word
256  uint32_t recordCount = recordNum - 1;
257  if (!byteOrder.isLocalEndian()) {
258  recordCount = SWAP_32(recordCount);
259  }
260  afChannel->seekg(FileHeader::RECORD_COUNT_OFFSET);
261  afChannel->write(reinterpret_cast<char *>(&recordCount), sizeof(recordCount));
262  if (afChannel->fail()) {
263  throw EvioException("error writing to file");
264  }
265  }
