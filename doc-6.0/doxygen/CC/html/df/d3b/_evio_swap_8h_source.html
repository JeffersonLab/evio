50  static void swapArray32(uint8_t *data, uint32_t dataLen) {
51 
52  if (data == nullptr || (dataLen % 4 != 0)) {
53  throw EvioException("bad arg");
54  }

56  uint32_t index;
57  uint8_t b1,b2,b3,b4;
58 
59  for (uint32_t i=0; i < dataLen; i+=4) {
60  index = i;
61 
62  b1 = (data[index]);
63  b2 = (data[index+1]);
64  b3 = (data[index+2]);
65  b4 = (data[index+3]);
66 
67  data[index] = b4;
68  data[index+1] = b3;
69  data[index+2] = b2;

107  static void swapEvent(std::shared_ptr<ByteBuffer> & srcBuffer,
108  std::shared_ptr<ByteBuffer> & destBuffer,
109  std::vector<std::shared_ptr<EvioNode>> & nodeList,
110  bool storeNodes=false, bool swapData=true,
111  size_t srcPos=0, size_t destPos=0) {
112 
113  if (srcBuffer == nullptr) {
114  throw EvioException("Null event in swapEvent");
115  }
116 
117  // Find the destination byte order and if it is to be swapped in place
118  bool inPlace = false;
119  ByteOrder srcOrder = srcBuffer->order();
120  ByteOrder destOrder = srcOrder.getOppositeEndian();
121 
122  if (destBuffer == nullptr || srcBuffer == destBuffer) {
123  // Do this so we can treat the source buffer as if it were a
124  // completely different destination buffer with its own byte order.
125  destBuffer = srcBuffer->duplicate();
126  destPos = srcPos;
127  inPlace = true;
128  }
129  destBuffer->order(destOrder);
130 
131  // Check position args
132  if (srcPos > srcBuffer->capacity() - 8) {
133  throw EvioException("bad value for srcPos arg");
134  }
135 
136  if (destPos > destBuffer->capacity() - 8) {
137  throw EvioException("bad value for destPos arg");
138  }
139 
140  // Create node
141  auto node = EvioNode::createEvioNode();
142 
143  // We know events are banks, so start by reading & swapping a bank header
144  swapBankHeader(node, srcBuffer, destBuffer, srcPos, destPos);
145 
146  // Store all nodes here
147  if (storeNodes) {
148  // Set a few special members for an event
149  node->eventNode = node;
150  node->scanned = true;
151  node->izEvent = true;
152  node->type = DataType::BANK.getValue();
153  nodeList.push_back(node);
154  }
155 // std::cout << "Created top node containing " << node->getDataTypeObj().toString() << std::endl;
156 
157  // The event is an evio bank so recursively swap it as such
158  swapStructure(node, srcBuffer, destBuffer, nodeList,
159  storeNodes, swapData, inPlace,
160  srcPos + 8, destPos + 8);
161 
162  if (inPlace) {
163  srcBuffer->order(destOrder);
164  }
165  }

184  static void swapBankHeader(std::shared_ptr<EvioNode> & node, std::shared_ptr<ByteBuffer> & srcBuffer,
185  std::shared_ptr<ByteBuffer> & destBuffer, size_t srcPos=0, size_t destPos=0) {
186 
187  try {
188  // Read & swap first bank header word
189  int length = srcBuffer->getInt(srcPos);
190  destBuffer->putInt(destPos, length);
191  srcPos += 4;
192  destPos += 4;
193 
194  // Read & swap second bank header word
195  uint32_t word = srcBuffer->getUInt(srcPos);
196  destBuffer->putInt(destPos, word);
197 
198  // node is never null
199  node->tag = word >> 16;
200 
201  uint32_t dt = (word >> 8) & 0xff;
202  node->dataType = dt & 0x3f;
203  node->pad = dt >> 6;
204 
205  // If only 7th bit set, it can be tag=0, num=0, type=0, padding=1.
206  // This regularly happens with composite data.
207  // However, it that MAY also be the legacy tagsegment type
208  // with no padding information. Ignore this as having tag & num
209  // in legacy code is probably rare.
210  //if (dt == 0x40) {
211  // node->dataType = DataType::TAGSEGMENT.getValue();
212  // node->pad = 0;
213  //}
214 
215  node->num = word & 0xff;
216  node->len = length;
217  node->pos = destPos - 4;
218  node->dataPos = destPos + 4;
219  node->dataLen = length - 1;
220  }
221  catch (std::overflow_error & e) {
222  throw EvioException("destBuffer too small to hold swapped data");
223  }
224  catch (std::underflow_error & e) {
225  throw EvioException("srcBuffer data underflow");
226  }
227  }

335  static void swapData(DataType const & type, std::shared_ptr<ByteBuffer> & srcBuffer,
336  std::shared_ptr<ByteBuffer> & destBuffer,
337  uint32_t len, bool inPlace = false,
338  size_t srcPos=0, size_t destPos=0) {
339 
340  // We end here
341  uint32_t endPos = srcPos + 4*len;
342 
343  // 64 bit swap
344  if (type == DataType::LONG64 ||
345  type == DataType::ULONG64 ||
346  type == DataType::DOUBLE64) {
347 
348  // When only swapping, no need to convert to double & back
349  for (; srcPos < endPos; srcPos += 8, destPos += 8) {
350  destBuffer->putLong(destPos, srcBuffer->getLong(srcPos));
351  }
352  }
353  // 32 bit swap
354  else if (type == DataType::INT32 ||
355  type == DataType::UINT32 ||
356  type == DataType::FLOAT32) {
357  // When only swapping, no need to convert to float & back
358  for (; srcPos < endPos; srcPos += 4, destPos += 4) {
359  destBuffer->putInt(destPos, srcBuffer->getInt(srcPos));
360  }
361  }
362  // 16 bit swap
363  else if (type == DataType::SHORT16 ||
364  type == DataType::USHORT16) {
365  for (; srcPos < endPos; srcPos += 2, destPos += 2) {
366  destBuffer->putShort(destPos, srcBuffer->getShort(srcPos));
367  }
368  }
369  // no swap
370  else if (type == DataType::UNKNOWN32 ||
371  type == DataType::CHAR8 ||
372  type == DataType::UCHAR8 ||
373  type == DataType::CHARSTAR8) {
374 
375  // 8 bit swap - no swap needed, but need to copy if destBuf != srcBuf
376  if (!inPlace) {
377  for (; srcPos < endPos; srcPos++, destPos++) {
378  destBuffer->put(destPos, srcBuffer->getByte(srcPos));
379  }
380  }
381  }
382  else if (type == DataType::COMPOSITE) {
383  // new composite type
384  CompositeData::swapAll(srcBuffer, destBuffer, srcPos, destPos, len);
385 
386  }
387  }

410  static void swapStructure(std::shared_ptr<EvioNode> & node,
411  std::shared_ptr<ByteBuffer> & srcBuffer,
412  std::shared_ptr<ByteBuffer> & destBuffer,
413  std::vector<std::shared_ptr<EvioNode>> & nodeList,
414  bool storeNodes=false, bool swapData=true, bool inPlace=false,
415  size_t srcPos=0, size_t destPos=0) {
416 
417  if (srcBuffer == nullptr || destBuffer == nullptr || node == nullptr) {
418  throw EvioException("arg(s) are null");
419  }
420 
421 // std::cout << "swapStructure: storeNodes = " << storeNodes << ", swapData = " << swapData <<
422 // ", inPlace = " << inPlace << ", srcPos = " << srcPos << ", destPos = " << destPos << std::endl;
423 
424  // Pass in header info through node object.
425  // Reuse this node object if not storing in a list of all nodes.
426  DataType dataType = node->getDataTypeObj();
427  uint32_t length = node->dataLen;
428 // std::cout << "swapStructure: dataType = " << dataType.toString() << ", len = " << length << std::endl;
429 
430  // If not a structure of structures, swap the data and return - no more recursion.
431  if (!dataType.isStructure()) {
432  // swap raw data here
433  if (swapData) EvioSwap::swapData(dataType, srcBuffer, destBuffer, length, inPlace, srcPos, destPos);
434  //std::cout << "HIT END_OF_LINE" << std::endl;
435  return;
436  }
437 
438  uint32_t offset = 0;
439  size_t sPos = srcPos, dPos = destPos;
440 
441  std::shared_ptr<EvioNode> firstNode = nullptr;
442  if (!nodeList.empty()) {
443  try {
444  firstNode = nodeList.at(0);
445  node = EvioNode::createEvioNode(firstNode);
446  } catch (const std::out_of_range &e) {
447  std::cerr << "Exception: " << e.what() << std::endl;
448  throw EvioException(e.what());
449  }
450  }
451 
452  // change 32-bit words to bytes
453  length *= 4;
454 
455  // This structure contains banks
456  if (dataType.isBank()) {
457  // Swap all banks contained in this structure
458  while (offset < length) {
459  swapBankHeader(node, srcBuffer, destBuffer, sPos, dPos);
460 
461  // Position offset to start of next header
462  offset += 4 * (node->len + 1); // plus 1 for length word
463 
464  // Recursive call, reuse node object if not storing
465  swapStructure(node, srcBuffer, destBuffer, nodeList,
466  storeNodes, swapData, inPlace,
467  sPos + 8, dPos + 8);
468 
469  sPos = srcPos + offset;
470  dPos = destPos + offset;
471 
472  // Store node objects
473  if (storeNodes) {
474  node->type = DataType::BANK.getValue();
475  nodeList.push_back(node);
476  if (offset < length) {
477  node = EvioNode::createEvioNode(firstNode);
478  }
479  }
480  }
481  }
482  else if (dataType.isSegment()) {
483  // extract all the segments from this bank.
484  while (offset < length) {
485  swapSegmentHeader(node, srcBuffer, destBuffer, sPos, dPos);
486 
487  offset += 4 * (node->len + 1);
488 
489  swapStructure(node, srcBuffer, destBuffer, nodeList,
490  storeNodes, swapData, inPlace,
491  sPos + 4, dPos + 4);
492 
493  sPos = srcPos + offset;
494  dPos = destPos + offset;
495 
496  // Store node objects
497  if (storeNodes) {
498  node->type = DataType::SEGMENT.getValue();
499  nodeList.push_back(node);
500  if (offset < length) {
501  node = EvioNode::createEvioNode(firstNode);
502  }
503  }
504  }
505  }
506  else if (dataType.isTagSegment()) {
507  // extract all the tag segments from this bank.
508  while (offset < length) {
509 
510  swapTagSegmentHeader(node, srcBuffer, destBuffer, sPos, dPos);
511 
512  offset += 4 * (node->len + 1);
513 
514  swapStructure(node, srcBuffer, destBuffer, nodeList,
515  storeNodes, swapData, inPlace,
516  sPos + 4, dPos + 4);
517 
518  sPos = srcPos + offset;
519  dPos = destPos + offset;
520 
521  if (storeNodes) {
522  node->type = DataType::TAGSEGMENT.getValue();
523  nodeList.push_back(node);
524  if (offset < length) {
525  node = EvioNode::createEvioNode(firstNode);
526  }
527  }
528  }
529  }
530  }

625  static void swapBank(uint32_t *buf, bool toLocal, uint32_t *dest) {
626  if (buf == nullptr) return;
627 
628  uint32_t dataLength, dataType;
629  uint32_t *p = buf;
630 
631  // Swap header to get length and contained type if buf NOT local endian
632  if (toLocal) {
633  p = ByteOrder::byteSwap32(buf, 2, dest);
634  }
635 
636  dataLength = p[0] - 1;
637  dataType = (p[1] >> 8) & 0x3f; // padding info in top 2 bits of type byte
638 std::cout << "swapBank: dataLen = " << dataLength << std::endl;
639 
640  // Swap header if buf is local endian
641  if (!toLocal) {
642  ByteOrder::byteSwap32(buf, 2, dest);
643  }
644 
645  // Swap non-header bank data
646  swapData(&buf[2], dataType, dataLength, toLocal, ((dest == nullptr) ? nullptr : &dest[2]));
647  }

834  static void swapData(uint32_t *src, uint32_t type, uint32_t length, bool toLocal, uint32_t *dst) {
835  if (src == nullptr) return;
836 
837  uint32_t fraglen, l = 0;
838 
839  // Swap the data or call swap_fragment
840  switch (type) {
841 
842  // 32-bit types: uint, float, or int
843  case 0x1:
844  case 0x2:
845  case 0xb:
846  ByteOrder::byteSwap32(src, length, dst);
847  break;
848 
849  // unknown or 8-bit types: string array, char, or uchar ... no swap
850  case 0x0:
851  case 0x3:
852  case 0x6:
853  case 0x7:
854  ByteOrder::byteNoSwap32(src, length, dst);
855  break;
856 
857  // 16-bit types: short or ushort
858  case 0x4:
859  case 0x5:
860  ByteOrder::byteSwap16(reinterpret_cast<uint16_t *>(src), 2 * length,
861  reinterpret_cast<uint16_t *>(dst));
862  break;
863 
864  // 64-bit types: double, int, or uint
865  case 0x8:
866  case 0x9:
867  case 0xa:
868  ByteOrder::byteSwap64(reinterpret_cast<uint64_t *>(src), length / 2,
869  reinterpret_cast<uint64_t *>(dst));
870  break;
871 
872  // Composite type
873  case 0xf:
874  CompositeData::swapAll(reinterpret_cast<uint8_t *>(src),
875  reinterpret_cast<uint8_t *>(dst),
876  length, !toLocal);
877  break;
878 
879  // Bank
880  case 0xe:
881  case 0x10:
882  while (l < length) {
883  // src is opposite local endian
884  if (toLocal) {
885  // swap bank
886  swapBank(&src[l], toLocal, (dst == nullptr) ? nullptr : &dst[l]);
887  // bank was this long (32 bit words) including header
888  fraglen = (dst == nullptr) ? src[l] + 1 : dst[l] + 1;
889  }
890  else {
891  fraglen = src[l] + 1;
892  swapBank(&src[l], toLocal, (dst == nullptr) ? nullptr : &dst[l]);
893  }
894  l += fraglen;
895  }
896  break;
897 
898  // Segment
899  case 0xd:
900  case 0x20:
901  while (l < length) {
902  if (toLocal) {
903  swapSegment(&src[l], toLocal, (dst == nullptr) ? nullptr : &dst[l]);
904  fraglen = (dst == nullptr) ? (src[l] & 0xffff) + 1 : (dst[l] & 0xffff) + 1;
905  }
906  else {
907  fraglen = (src[l] & 0xffff) + 1;
908  swapSegment(&src[l], toLocal, (dst == nullptr) ? nullptr : &dst[l]);
909  }
910  l += fraglen;
911  }
912  break;
913 
914  // Tagsegment
915  case 0xc:
916  while (l < length) {
917  if (toLocal) {
918  swapTagsegment(&src[l], toLocal, (dst == nullptr) ? nullptr : &dst[l]);
919  fraglen = (dst == nullptr) ? (src[l] & 0xffff) + 1 : (dst[l] & 0xffff) + 1;
920  }
921  else {
922  fraglen = (src[l] & 0xffff) + 1;
923  swapTagsegment(&src[l], toLocal, (dst == nullptr) ? nullptr : &dst[l]);
924  }
925  l += fraglen;
926  }
927  break;
928 
929  // unknown type, just copy
930  default:
931  ByteOrder::byteNoSwap32(src, length, dst);
932  break;
933  }
934  }

951  static void swapData(std::shared_ptr<BaseStructure> strc) {
952  auto type = strc->getHeader()->getDataType();
953  uint32_t length = strc->getHeader()->getDataLength();
954 
955  if (type == DataType::UINT32) {
956  auto & vec = strc->getUIntData();
957  ByteOrder::byteSwap32(vec.data(), length, nullptr);
958  strc->updateUIntData();
959  }
960  else if (type == DataType::INT32) {
961  auto & vec = strc->getIntData();
962  ByteOrder::byteSwap32(reinterpret_cast<uint32_t*>(vec.data()), length, nullptr);
963  strc->updateIntData();
964  }
