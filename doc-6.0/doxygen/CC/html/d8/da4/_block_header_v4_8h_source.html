566  static uint32_t generateSixthWord(std::bitset<24> const & bSet, bool hasDictionary, bool isEnd) {
567  uint32_t v = 4; // version
568 
569  for (uint32_t i=0; i < bSet.size(); i++) {

597  static uint32_t generateSixthWord(uint32_t version, bool hasDictionary,
598  bool isEnd, uint32_t eventType) {
599  uint32_t v = version;
