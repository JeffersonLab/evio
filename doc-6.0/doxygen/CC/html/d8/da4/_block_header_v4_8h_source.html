564  static uint32_t generateSixthWord(std::bitset<24> const & bSet, uint32_t version = 4,
565  bool hasDictionary = false,
566  bool isEnd = false, uint32_t eventType = 0,
567  bool hasFirstEv = false, bool isStreaming = false) {
568 
569  uint32_t v = version;
570 
571  for (size_t i=0; i < bSet.size(); i++) {
572  if (i > 23) {
573  break;
574  }
575  if (bSet[i]) {
576  v |= (0x1 << (8+i));
577  }
578  }
579 
580  v = hasDictionary ? (v | 0x100) : v;
581  v = isEnd ? (v | 0x200) : v;
582  v |= ((eventType & 0xf) << 10);
583  v = hasFirstEv ? (v | 0x4000) : v;
584  v = isStreaming ? (v | 0x8000) : v;
585 
586  return v;
587  }
