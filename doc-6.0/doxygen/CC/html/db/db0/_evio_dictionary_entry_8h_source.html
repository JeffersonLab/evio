295  bool operator==(const EvioDictionaryEntry &other) const {
296 
297  if (&other == this) {
298  return true;

300 
301  // Objects equal each other if tag & num & tagEnd are the same
302  auto otherParent = other.getParentEntry();
303 
304  bool match = (tag == other.tag);
305 
306  if (numValid != other.numValid) {
307  return false;
308  }
309  match = match && (num == other.num);
310 
311  // Now check tag range if any
312  match = match && (tagEnd == other.tagEnd);
313 
314  // Now check if same entry type
315  match = match && (entryType == other.entryType);
316 
317  // If both parent containers are defined, use them as well
318  if (parentEntry != nullptr && otherParent != nullptr) {
