445  static uint32_t findEvioVersion(ByteBuffer & bb, size_t initialPos) {
446  // Look at first record header
447 
448  // Have enough remaining bytes to read 8 words of header?
449  if (bb.limit() - initialPos < 32) {
450  throw std::underflow_error("not enough data to read in header");
451  }
452 
453  // Set the byte order to match the file's ordering.
454 
455  // Check the magic number for endianness (buffer defaults to big endian)
456  ByteOrder byteOrder = bb.order();
457 
458  // Offset to magic # is in the SAME LOCATION FOR ALL EVIO VERSIONS
459  uint32_t magicNumber = bb.getUInt(initialPos + IBlockHeader::MAGIC_OFFSET);
460  if (magicNumber != IBlockHeader::MAGIC_NUMBER) {
461  if (byteOrder == ByteOrder::ENDIAN_BIG) {
462  byteOrder = ByteOrder::ENDIAN_LITTLE;
463  }
464  else {
465  byteOrder = ByteOrder::ENDIAN_BIG;
466  }
467  bb.order(byteOrder);
468 
469  // Reread magic number to make sure things are OK
470  magicNumber = bb.getInt(initialPos + IBlockHeader::MAGIC_OFFSET);
471  if (magicNumber != IBlockHeader::MAGIC_NUMBER) {
472  throw EvioException("magic number is bad, " + std::to_string(magicNumber));
473  }
474  }
475 
476  // Find the version number, again, SAME LOCATION FOR ALL EVIO VERSIONS
477  uint32_t bitInfo = bb.getUInt(initialPos + IBlockHeader::BIT_INFO_OFFSET);
478  return bitInfo & IBlockHeader::VERSION_MASK;
479  }

700  static int powerOfTwo(int x, bool roundUp) {
701  if (x < 0) return -1;
702 
703  // The following algorithm is found in
704  // "Hacker's Delight" by Henry Warren Jr.
705 
706  if (roundUp) {
707  x = x - 1;
708  x |= (x>>1);
709  x |= (x>>2);
710  x |= (x>>4);
711  x |= (x>>8);
712  x |= (x>>16);
713  return x + 1;
714  }
715 
716  int y;
717  do {
718  y = x;
719  x &= (x - 1);
720  } while (x != 0);
721  return y;
722  }

775  static size_t stringsToRawSize(std::vector<std::string> const & strings) {
776 
777  if (strings.empty()) {
778  return 0;
779  }
780 
781  size_t dataLen = 0;
782  for (std::string const & s : strings) {
783  dataLen += s.length() + 1; // don't forget the null char after each string
784  }
785 
786  // Add any necessary padding to 4 byte boundaries.
787  // IMPORTANT: There must be at least one '\004'
788  // character at the end. This distinguishes evio
789  // string array version from earlier version.
790  int pads[] = {4,3,2,1};
791  dataLen += pads[dataLen%4];
792 
793  return dataLen;
794  }

830  static void stringsToRawBytes(std::vector<std::string> & strings,
831  std::vector<uint8_t> & bytes) {
832 
833  if (strings.empty()) {
834  bytes.clear();
835  return;
836  }
837 
838  // create some storage
839  size_t dataLen = stringsToRawSize(strings);
840  std::string strData;
841  strData.reserve(dataLen);
842 
843  for (std::string const & s : strings) {
844  // add string
845  strData.append(s);
846  // add ending null
847  strData.append(1, '\000');
848  }
849 
850  // Add any necessary padding to 4 byte boundaries.
851  // IMPORTANT: There must be at least one '\004'
852  // character at the end. This distinguishes evio
853  // string array version from earlier version.
854  int pads[] = {4,3,2,1};
855  switch (pads[strData.length()%4]) {
856  case 4:
857  strData.append(4, '\004');
858  break;
859  case 3:
860  strData.append(3, '\004');
861  break;
862  case 2:
863  strData.append(2, '\004');
864  break;
865  case 1:
866  strData.append(1, '\004');
867  }
868 
869  // Transform to ASCII
870  bytes.resize(dataLen);
871  for (size_t i=0; i < strData.length(); i++) {
872  bytes[i] = strData[i];
873  }
874  }

981  static void stringBuilderToStrings(std::string const & strData, bool onlyGoodChars,
982  std::vector<std::string> & strings) {
983 
984  // Each string is terminated with a null (char val = 0)
985  // and in addition, the end is padded by ASCII 4's (char val = 4).
986  // However, in the legacy versions of evio, there is only one
987  // null-terminated string and anything as padding. To accommodate legacy evio, if
988  // there is not an ending ASCII value 4, anything past the first null is ignored.
989  // After doing so, split at the nulls. Do not use the String
990  // method "split" as any empty trailing strings are unfortunately discarded.
991 
992  char c;
993  std::vector<int> nullIndexList;
994  nullIndexList.reserve(10);
995  uint32_t nullCount = 0, goodChars = 0;
996  bool badFormat = true;
997 
998  size_t length = strData.length();
999  bool noEnding4 = false;
1000  if (strData[length - 1] != '\004') {
1001  noEnding4 = true;
1002  }
1003 
1004  for (int i=0; i < length; i++) {
1005  c = strData[i];
1006 
1007  // If char is a null
1008  if (c == 0) {
1009  nullCount++;
1010  nullIndexList.push_back(i);
1011  // If evio v2 or 3, only 1 null terminated string exists
1012  // and padding is just junk or nonexistent.
1013  if (noEnding4) {
1014  badFormat = false;
1015  break;
1016  }
1017  }
1018  // Look for any non-printing/control characters (not including null)
1019  // and end the string there. Allow tab & newline.
1020  else if ((c < 32 || c > 126) && c != 9 && c != 10) {
1021  if (nullCount < 1) {
1022  badFormat = true;
1023  // Getting garbage before first null.
1024  break;
1025  }
1026 
1027  // Already have at least one null & therefore a String.
1028  // Now we have junk or non-printing ascii which is
1029  // possibly the ending 4.
1030 
1031  // If we have a 4, investigate further to see if format
1032  // is entirely valid.
1033  if (c == '\004') {
1034  // How many more chars are there?
1035  size_t charsLeft = length - (i+1);
1036 
1037  // Should be no more than 3 additional 4's before the end
1038  if (charsLeft > 3) {
1039  badFormat = true;
1040  break;
1041  }
1042  else {
1043  // Check to see if remaining chars are all 4's. If not, bad.
1044  for (int j=1; j <= charsLeft; j++) {
1045  c = strData[i+j];
1046  if (c != '\004') {
1047  badFormat = true;
1048  goto pastOuterLoop;
1049  }
1050  }
1051  badFormat = false;
1052  break;
1053  }
1054  }
1055  else {
1056  badFormat = true;
1057  break;
1058  }
1059  }
1060 
1061  pastOuterLoop:
1062 
1063  // Number of good ASCII chars we have
1064  goodChars++;
1065  }
1066 
1067  strings.clear();
1068 
1069  if (badFormat) {
1070  if (onlyGoodChars) {
1071  // Return everything in one String WITHOUT garbage
1072  std::string goodStr(strData.data(), goodChars);
1073  strings.push_back(goodStr);
1074  return;
1075  }
1076  // Return everything in one String including possible garbage
1077  strings.push_back(strData);
1078  return;
1079  }
1080 
1081  // If here, raw bytes are in the proper format
1082 
1083  int firstIndex = 0;
1084  for (int nullIndex : nullIndexList) {
1085  std::string str(strData.data() + firstIndex, (nullIndex - firstIndex));
1086  strings.push_back(str);
1087  firstIndex = nullIndex + 1;
1088  }
1089  }

1124  static uint32_t countAndFixIntSpecifiers(std::string & text) {
1125  static std::regex specifier("%(\\d*)([xd])");
1126 
1127  auto begin = std::sregex_iterator(text.begin(), text.end(), specifier);
1128  auto end = std::sregex_iterator();
1129  uint32_t specifierCount = std::distance(begin, end);
1130 
1131  std::sregex_iterator i = begin;
1132 
1133  // Go thru all specifiers in text, only once
1134  for (uint32_t j = 0; j < specifierCount; j++) {
1135  if (j > 0) {
1136  // skip over specifiers previously dealt with (text can change each loop)
1137  i = std::sregex_iterator(text.begin(), text.end(), specifier);

1140  }

1148  // Does not compile on linux gcc
1149  // text.replace(match[1].first, match[1].second, "0" + specWidth);

1152  // This first sub match occurs at the beginning of the match

1157  }
