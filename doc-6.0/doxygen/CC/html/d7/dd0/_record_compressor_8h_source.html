135  void run() {
136 
137  try {
138 
139  // The first time through, we need to release all records coming before
140  // our first in case there are < threadNumber records before close() is called.
141  // This way close() is not waiting for thread #12 to get and subsequently
142  // release items 0 - 11 when there were only 5 records total.
143  // (threadNumber starts at 0).
144 
145  // Be careful when dealing with negative numbers and unsigned ints ...
146  int64_t seqNumber = (int64_t)threadNumber - 1;
147  supply->release(threadNumber, seqNumber);
148 
149  while (true) {
150 
151  // Get the next record for this thread to compress
152  auto item = supply->getToCompress(threadNumber);
153 
154  {
155  // Only allow interruption when blocked on trying to get item
156  boost::this_thread::disable_interruption d1;
157 
158  // Pull record out of wrapping object
159  std::shared_ptr<RecordOutput> record = item->getRecord();
160  // Set compression type
161  auto header = record->getHeader();
162  header->setCompressionType(compressionType);
163 //cout << "RecordCompressor thd " << threadNumber << ": got record, set rec # to " << header->getRecordNumber() << endl;
164  // Do compression
165  record->build();
166  // Release back to supply
167  supply->releaseCompressor(item);
168  }
169  }
170  }
171  catch (Disruptor::AlertException & e) {
172  // Woken up in getToCompress through user call to supply.errorAlert()
173  //std::cout << "RecordCompressor: quit thread through alert" << std::endl;
174  }
175  catch (boost::thread_interrupted & e) {
176  // Never happen since we don't use timeout wait strategy
177  //cout << "RecordCompressor thd " << threadNumber << ": INTERRUPTED, return" << endl;
178  }
179  }
180  };
