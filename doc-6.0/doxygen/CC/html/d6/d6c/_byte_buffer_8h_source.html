223  template<typename T> T read(size_t index) const {
224  if (index + sizeof(T) <= lim) {
225  return *((T *) &(buf.get())[index + off]);

232  template<typename T> void write(T & data) {
233  size_t s = sizeof(data);

235  if (lim < (pos + s)) {
236  // Write would exceeded limit

239  memcpy((void *) (&(buf.get())[pos + off]), (void *) (&data), s);
