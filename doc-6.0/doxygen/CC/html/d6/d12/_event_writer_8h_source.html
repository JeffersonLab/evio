266  void run() {
267 
268  try {
269 
270  while (true) {
271 
272  // Get the next record for this thread to write
273  // shared_ptr<RecordRingItem>
274  auto item = supply->getToWrite();
275 
276  {
277  // Only allow interruption when blocked on trying to get item
278  boost::this_thread::disable_interruption d1;
279 
280  int64_t currentSeq = item->getSequence();
281  // Pull record out of wrapping object
282  auto record = item->getRecord();
283 
284  // Only need to check the disk when writing the first record following
285  // a file split. That first write will create the file. If there isn't
286  // enough room, then flag will be set.
287  bool checkDisk = item->isCheckDisk();
288 
289  // Check the disk before we try to write if about to create another file,
290  // we're told to check the disk, and we're not forcing to disk
291  if ((writer->bytesWritten < 1) && checkDisk && (!forceToDisk.load())) {
292 
293  // If there isn't enough free space to write the complete, projected
294  // size file, and we're not trying to force the write ...
295  // store a COPY of the record for now and release the original so
296  // that writeEventToFile() does not block.
297  //
298  // So here is the problem. We are stuck in a loop here if disk is full.
299  // If events are flowing and since writing data to file is the bottleneck,
300  // it is likely that all records have been filled and published onto
301  // the ring. AND, writeEventToFile() blocks in a spin as it tries to get the
302  // next record from the ring which, unfortunately, never comes.
303  //
304  // When writeEventToFile() blocks, it can't respond by returning a "false" value.
305  // This plays havoc with code like the emu which is not expecting the write
306  // to block (at least not for very long).
307  //
308  // To break writeEventToFile() out of its spinning block, we make a copy of the
309  // item we're trying to write and release the original record. This allows
310  // writeEventToFile() to grab a new (newly released) record, write an event into
311  // it, and return to the caller. From then on, writeEventToFile() can prevent
312  // blocking by checking for a full disk (which it couldn't do before since
313  // the signal came too late).
314 
315  while (writer->fullDisk() && (!forceToDisk.load())) {
316  // Wait for a sec and try again
317  std::this_thread::sleep_for(std::chrono::seconds(1));
318 
319  // If we released the item in a previous loop, don't do it again
320  if (!item->isAlreadyReleased()) {
321  // Copy item
322  auto copiedItem = storeRecordCopy(item);
323  // Release original so we don't block writeEvent()
324  supply->releaseWriter(item);
325  item = copiedItem;
326  }
327 
328  // Wait until space opens up
329  }
330 
331  // If we're here, there must be free space available even
332  // if there previously wasn't.
333  }
334 
335  // Do write
336  // Write current item to file
337  //cout << "EventWriter: Calling writeToFileMT(item)\n";
338  writer->writeToFileMT(item, forceToDisk.load());
339 
340  // Turn off forced write to disk, if the record which
341  // initially triggered it, has now been written.
342  if (forceToDisk.load() && (forcedRecordId.load() == item->getId())) {
343  //cout << "EventWriter: WROTE the record that triggered force, reset to false\n";
344  forceToDisk = false;
345  }
346 
347  // Now we're done with this sequence
348  lastSeqProcessed = currentSeq;
349 
350  // Split file if needed
351  if (item->splitFileAfterWrite()) {
352  writer->splitFile();
353  }
354 
355  // Release back to supply
356  supply->releaseWriter(item);
357  }
358  }
359  }
360  catch (boost::thread_interrupted & e) {
361  std::cout << "EventWriter: INTERRUPTED, return" << std::endl;
362  }
363  }
